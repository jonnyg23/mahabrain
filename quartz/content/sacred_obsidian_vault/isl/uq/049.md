 <h1 align="CENTER">49. ‏الحجرات‎</h1>
 <h2 align="CENTER">'lḥjr't</h2>
 <table width="100%" border="1">
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_049_001">1</a> ‏يَٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ لَا تُقَدِّمُوا۟ بَيْنَ يَدَىِ ٱللَّهِ وَرَسُولِهِۦ ۖ وَٱتَّقُوا۟ ٱللَّهَ ۚ إِنَّ ٱللَّهَ سَمِيعٌ عَلِيمٌۭ ‎</p>
 </td>
 <td width="50%">yaʔ'ʔay:uha' 'l:aðiyna ʔa'manuw' la' tuqad:imuw' bayna yadayi 'll:ahi warasuwlihiy sl' wa't:aquw' 'll:aha j 'ʔin:a 'll:aha samiyʕun ʕaliymunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_049_002">2</a> ‏يَٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ لَا تَرْفَعُوٓا۟ أَصْوَتَكُمْ فَوْقَ صَوْتِ ٱلنَّبِىِّ وَلَا تَجْهَرُوا۟ لَهُۥ بِٱلْقَوْلِ كَجَهْرِ بَعْضِكُمْ لِبَعْضٍ أَن تَحْبَطَ أَعْمَلُكُمْ وَأَنتُمْ لَا تَشْعُرُونَ ‎</p>
 </td>
 <td width="50%">yaʔ'ʔay:uha' 'l:aðiyna ʔa'manuw' la' tarfaʕuwʔ' 'ʔaṣwatakum fawqa ṣawti 'ln:abiy:i wala' tajharuw' lahuw bi'lqawli kajahri baʕḍikum libaʕḍin 'ʔan taḥbaṭa 'ʔaʕmalukum wa'ʔantum la' taʃʕuruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_049_003">3</a> ‏إِنَّ ٱلَّذِينَ يَغُضُّونَ أَصْوَتَهُمْ عِندَ رَسُولِ ٱللَّهِ أُو۟لَٓئِكَ ٱلَّذِينَ ٱمْتَحَنَ ٱللَّهُ قُلُوبَهُمْ لِلتَّقْوَىٰ ۚ لَهُم مَّغْفِرَةٌۭ وَأَجْرٌ عَظِيمٌ ‎</p>
 </td>
 <td width="50%">'ʔin:a 'l:aðiyna yaɣuḍ:uwna 'ʔaṣwatahum ʕinda rasuwli 'll:ahi 'ʔuwlaʔyʔika 'l:aðiyna 'mtaḥana 'll:ahu quluwbahum lilt:aqway' j lahum m:aɣfiratunm wa'ʔajrun ʕaẓiymun </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_049_004">4</a> ‏إِنَّ ٱلَّذِينَ يُنَادُونَكَ مِن وَرَآءِ ٱلْحُجُرَتِ أَكْثَرُهُمْ لَا يَعْقِلُونَ ‎</p>
 </td>
 <td width="50%">'ʔin:a 'l:aðiyna yuna'duwnaka min wara'ʔʔi 'lḥujurati 'ʔakθaruhum la' yaʕqiluwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_049_005">5</a> ‏وَلَوْ أَنَّهُمْ صَبَرُوا۟ حَتَّىٰ تَخْرُجَ إِلَيْهِمْ لَكَانَ خَيْرًۭا لَّهُمْ ۚ وَٱللَّهُ غَفُورٌۭ رَّحِيمٌۭ ‎</p>
 </td>
 <td width="50%">walaw 'ʔan:ahum ṣabaruw' ḥat:ay' taxruja 'ʔilayhim laka'na xayranm' l:ahum j wa'll:ahu ɣafuwrunm r:aḥiymunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_049_006">6</a> ‏يَٓأَيُّهَا ٱلَّذِينَ ءَامَنُوٓا۟ إِن جَآءَكُمْ فَاسِقٌۢ بِنَبَإٍۢ فَتَبَيَّنُوٓا۟ أَن تُصِيبُوا۟ قَوْمًۢا بِجَهَلَةٍۢ فَتُصْبِحُوا۟ عَلَىٰ مَا فَعَلْتُمْ نَدِمِينَ ‎</p>
 </td>
 <td width="50%">yaʔ'ʔay:uha' 'l:aðiyna ʔa'manuwʔ' 'ʔin ja'ʔʔakum fa'siqunm binaba'ʔinm fatabay:anuwʔ' 'ʔan tuṣiybuw' qawmanm' bijahalatinm fatuṣbiḥuw' ʕalay' ma' faʕaltum nadimiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_049_007">7</a> ‏وَٱعْلَمُوٓا۟ أَنَّ فِيكُمْ رَسُولَ ٱللَّهِ ۚ لَوْ يُطِيعُكُمْ فِى كَثِيرٍۢ مِّنَ ٱلْأَمْرِ لَعَنِتُّمْ وَلَكِنَّ ٱللَّهَ حَبَّبَ إِلَيْكُمُ ٱلْإِيمَنَ وَزَيَّنَهُۥ فِى قُلُوبِكُمْ وَكَرَّهَ إِلَيْكُمُ ٱلْكُفْرَ وَٱلْفُسُوقَ وَٱلْعِصْيَانَ ۚ أُو۟لَٓئِكَ هُمُ ٱلرَّشِدُونَ ‎</p>
 </td>
 <td width="50%">wa'ʕlamuwʔ' 'ʔan:a fiykum rasuwla 'll:ahi j law yuṭiyʕukum fiy kaθiyrinm m:ina 'l'ʔamri laʕanit:um walakin:a 'll:aha ḥab:aba 'ʔilaykumu 'l'ʔiymana wazay:anahuw fiy quluwbikum wakar:aha 'ʔilaykumu 'lkufra wa'lfusuwqa wa'lʕiṣya'na j 'ʔuwlaʔyʔika humu 'lr:aʃiduwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_049_008">8</a> ‏فَضْلًۭا مِّنَ ٱللَّهِ وَنِعْمَةًۭ ۚ وَٱللَّهُ عَلِيمٌ حَكِيمٌۭ ‎</p>
 </td>
 <td width="50%">faḍlanm' m:ina 'll:ahi waniʕmatanm j wa'll:ahu ʕaliymun ḥakiymunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_049_009">9</a> ‏وَإِن طَآئِفَتَانِ مِنَ ٱلْمُؤْمِنِينَ ٱقْتَتَلُوا۟ فَأَصْلِحُوا۟ بَيْنَهُمَا ۖ فَإِنۢ بَغَتْ إِحْدَىٰهُمَا عَلَى ٱلْأُخْرَىٰ فَقَتِلُوا۟ ٱلَّتِى تَبْغِى حَتَّىٰ تَفِىٓءَ إِلَىٰٓ أَمْرِ ٱللَّهِ ۚ فَإِن فَآءَتْ فَأَصْلِحُوا۟ بَيْنَهُمَا بِٱلْعَدْلِ وَأَقْسِطُوٓا۟ ۖ إِنَّ ٱللَّهَ يُحِبُّ ٱلْمُقْسِطِينَ ‎</p>
 </td>
 <td width="50%">wa'ʔin ṭa'ʔyʔifata'ni mina 'lmuwʔminiyna 'qtataluw' fa'ʔaṣliḥuw' baynahuma' sl' fa'ʔinm baɣat 'ʔiḥday'huma' ʕalay 'l'ʔuxray' faqatiluw' 'l:atiy tabɣiy ḥat:ay' tafiyʔʔa 'ʔilay'ʔ 'ʔamri 'll:ahi j fa'ʔin fa'ʔʔat fa'ʔaṣliḥuw' baynahuma' bi'lʕadli wa'ʔaqsiṭuwʔ' sl' 'ʔin:a 'll:aha yuḥib:u 'lmuqsiṭiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_049_010">10</a> ‏إِنَّمَا ٱلْمُؤْمِنُونَ إِخْوَةٌۭ فَأَصْلِحُوا۟ بَيْنَ أَخَوَيْكُمْ ۚ وَٱتَّقُوا۟ ٱللَّهَ لَعَلَّكُمْ تُرْحَمُونَ ‎</p>
 </td>
 <td width="50%">'ʔin:ama' 'lmuwʔminuwna 'ʔixwatunm fa'ʔaṣliḥuw' bayna 'ʔaxawaykum j wa't:aquw' 'll:aha laʕal:akum turḥamuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_049_011">11</a> ‏يَٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ لَا يَسْخَرْ قَوْمٌۭ مِّن قَوْمٍ عَسَىٰٓ أَن يَكُونُوا۟ خَيْرًۭا مِّنْهُمْ وَلَا نِسَآءٌۭ مِّن نِّسَآءٍ عَسَىٰٓ أَن يَكُنَّ خَيْرًۭا مِّنْهُنَّ ۖ وَلَا تَلْمِزُوٓا۟ أَنفُسَكُمْ وَلَا تَنَابَزُوا۟ بِٱلْأَلْقَبِ ۖ بِئْسَ ٱلِٱسْمُ ٱلْفُسُوقُ بَعْدَ ٱلْإِيمَنِ ۚ وَمَن لَّمْ يَتُبْ فَأُو۟لَٓئِكَ هُمُ ٱلظَّلِمُونَ ‎</p>
 </td>
 <td width="50%">yaʔ'ʔay:uha' 'l:aðiyna ʔa'manuw' la' yasxar qawmunm m:in qawmin ʕasay'ʔ 'ʔan yakuwnuw' xayranm' m:inhum wala' nisa'ʔʔunm m:in n:isa'ʔʔin ʕasay'ʔ 'ʔan yakun:a xayranm' m:inhun:a sl' wala' talmizuwʔ' 'ʔanfusakum wala' tana'bazuw' bi'l'ʔalqabi sl' biyʔsa 'li'smu 'lfusuwqu baʕda 'l'ʔiymani j waman l:am yatub fa'ʔuwlaʔyʔika humu 'lẓ:alimuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_049_012">12</a> ‏يَٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ ٱجْتَنِبُوا۟ كَثِيرًۭا مِّنَ ٱلظَّنِّ إِنَّ بَعْضَ ٱلظَّنِّ إِثْمٌۭ ۖ وَلَا تَجَسَّسُوا۟ وَلَا يَغْتَب بَّعْضُكُم بَعْضًا ۚ أَيُحِبُّ أَحَدُكُمْ أَن يَأْكُلَ لَحْمَ أَخِيهِ مَيْتًۭا فَكَرِهْتُمُوهُ ۚ وَٱتَّقُوا۟ ٱللَّهَ ۚ إِنَّ ٱللَّهَ تَوَّابٌۭ رَّحِيمٌۭ ‎</p>
 </td>
 <td width="50%">yaʔ'ʔay:uha' 'l:aðiyna ʔa'manuw' 'jtanibuw' kaθiyranm' m:ina 'lẓ:an:i 'ʔin:a baʕḍa 'lẓ:an:i 'ʔiθmunm sl' wala' tajas:asuw' wala' yaɣtab b:aʕḍukum baʕḍan' j 'ʔayuḥib:u 'ʔaḥadukum 'ʔan ya'ʔkula laḥma 'ʔaxiyhi maytanm' fakarihtumuwhu j wa't:aquw' 'll:aha j 'ʔin:a 'll:aha taw:a'bunm r:aḥiymunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_049_013">13</a> ‏يَٓأَيُّهَا ٱلنَّاسُ إِنَّا خَلَقْنَكُم مِّن ذَكَرٍۢ وَأُنثَىٰ وَجَعَلْنَكُمْ شُعُوبًۭا وَقَبَآئِلَ لِتَعَارَفُوٓا۟ ۚ إِنَّ أَكْرَمَكُمْ عِندَ ٱللَّهِ أَتْقَىٰكُمْ ۚ إِنَّ ٱللَّهَ عَلِيمٌ خَبِيرٌۭ ‎</p>
