 <h1 align="CENTER">14. ‏إبراهيم‎</h1>
 <h2 align="CENTER">'ʔbr'hym</h2>
 <table width="100%" border="1">
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_014_001">1</a> ‏الٓر ۚ كِتَبٌ أَنزَلْنَهُ إِلَيْكَ لِتُخْرِجَ ٱلنَّاسَ مِنَ ٱلظُّلُمَتِ إِلَى ٱلنُّورِ بِإِذْنِ رَبِّهِمْ إِلَىٰ صِرَطِ ٱلْعَزِيزِ ٱلْحَمِيدِ ‎</p>
 </td>
 <td width="50%">'lʔr j kitabun 'ʔanzalnahu 'ʔilayka lituxrija 'ln:a'sa mina 'lẓ:ulumati 'ʔilay 'ln:uwri bi'ʔiðni rab:ihim 'ʔilay' ṣiraṭi 'lʕaziyzi 'lḥamiydi </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_014_002">2</a> ‏ٱللَّهِ ٱلَّذِى لَهُۥ مَا فِى ٱلسَّمَوَتِ وَمَا فِى ٱلْأَرْضِ ۗ وَوَيْلٌۭ لِّلْكَفِرِينَ مِنْ عَذَابٍۢ شَدِيدٍ ‎</p>
 </td>
 <td width="50%">'ll:ahi 'l:aðiy lahuw ma' fiy 'ls:amawati wama' fiy 'l'ʔarḍi ql' wawaylunm l:ilkafiriyna min ʕaða'binm ʃadiydin </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_014_003">3</a> ‏ٱلَّذِينَ يَسْتَحِبُّونَ ٱلْحَيَوٰةَ ٱلدُّنْيَا عَلَى ٱلْءَاخِرَةِ وَيَصُدُّونَ عَن سَبِيلِ ٱللَّهِ وَيَبْغُونَهَا عِوَجًا ۚ أُو۟لَٓئِكَ فِى ضَلَلٍۭ بَعِيدٍۢ ‎</p>
 </td>
 <td width="50%">'l:aðiyna yastaḥib:uwna 'lḥayaw'ta 'ld:unya' ʕalay 'lʔa'xirati wayaṣud:uwna ʕan sabiyli 'll:ahi wayabɣuwnaha' ʕiwajan' j 'ʔuwlaʔyʔika fiy ḍalalinm baʕiydinm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_014_004">4</a> ‏وَمَآ أَرْسَلْنَا مِن رَّسُولٍ إِلَّا بِلِسَانِ قَوْمِهِۦ لِيُبَيِّنَ لَهُمْ ۖ فَيُضِلُّ ٱللَّهُ مَن يَشَآءُ وَيَهْدِى مَن يَشَآءُ ۚ وَهُوَ ٱلْعَزِيزُ ٱلْحَكِيمُ ‎</p>
 </td>
 <td width="50%">wama'ʔ 'ʔarsalna' min r:asuwlin 'ʔil:a' bilisa'ni qawmihiy liyubay:ina lahum sl' fayuḍil:u 'll:ahu man yaʃa'ʔʔu wayahdiy man yaʃa'ʔʔu j wahuwa 'lʕaziyzu 'lḥakiymu </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_014_005">5</a> ‏وَلَقَدْ أَرْسَلْنَا مُوسَىٰ بِـَٔايَتِنَآ أَنْ أَخْرِجْ قَوْمَكَ مِنَ ٱلظُّلُمَتِ إِلَى ٱلنُّورِ وَذَكِّرْهُم بِأَيَّىٰمِ ٱللَّهِ ۚ إِنَّ فِى ذَلِكَ لَءَايَتٍۢ لِّكُلِّ صَبَّارٍۢ شَكُورٍۢ ‎</p>
 </td>
 <td width="50%">walaqad 'ʔarsalna' muwsay' biʔa'yatina'ʔ 'ʔan 'ʔaxrij qawmaka mina 'lẓ:ulumati 'ʔilay 'ln:uwri waðak:irhum bi'ʔay:ay'mi 'll:ahi j 'ʔin:a fiy ðalika laʔa'yatinm l:ikul:i ṣab:a'rinm ʃakuwrinm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_014_006">6</a> ‏وَإِذْ قَالَ مُوسَىٰ لِقَوْمِهِ ٱذْكُرُوا۟ نِعْمَةَ ٱللَّهِ عَلَيْكُمْ إِذْ أَنجَىٰكُم مِّنْ ءَالِ فِرْعَوْنَ يَسُومُونَكُمْ سُوٓءَ ٱلْعَذَابِ وَيُذَبِّحُونَ أَبْنَآءَكُمْ وَيَسْتَحْيُونَ نِسَآءَكُمْ ۚ وَفِى ذَلِكُم بَلَآءٌۭ مِّن رَّبِّكُمْ عَظِيمٌۭ ‎</p>
 </td>
 <td width="50%">wa'ʔið qa'la muwsay' liqawmihi 'ðkuruw' niʕmata 'll:ahi ʕalaykum 'ʔið 'ʔanjay'kum m:in ʔa'li firʕawna yasuwmuwnakum suwʔʔa 'lʕaða'bi wayuðab:iḥuwna 'ʔabna'ʔʔakum wayastaḥyuwna nisa'ʔʔakum j wafiy ðalikum bala'ʔʔunm m:in r:ab:ikum ʕaẓiymunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_014_007">7</a> ‏وَإِذْ تَأَذَّنَ رَبُّكُمْ لَىِٕن شَكَرْتُمْ لَأَزِيدَنَّكُمْ ۖ وَلَىِٕن كَفَرْتُمْ إِنَّ عَذَابِى لَشَدِيدٌۭ ‎</p>
 </td>
 <td width="50%">wa'ʔið ta'ʔað:ana rab:ukum layʔin ʃakartum la'ʔaziydan:akum sl' walayʔin kafartum 'ʔin:a ʕaða'biy laʃadiydunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_014_008">8</a> ‏وَقَالَ مُوسَىٰٓ إِن تَكْفُرُوٓا۟ أَنتُمْ وَمَن فِى ٱلْأَرْضِ جَمِيعًۭا فَإِنَّ ٱللَّهَ لَغَنِىٌّ حَمِيدٌ ‎</p>
 </td>
 <td width="50%">waqa'la muwsay'ʔ 'ʔin takfuruwʔ' 'ʔantum waman fiy 'l'ʔarḍi jamiyʕanm' fa'ʔin:a 'll:aha laɣaniy:un ḥamiydun </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_014_009">9</a> ‏أَلَمْ يَأْتِكُمْ نَبَؤُا۟ ٱلَّذِينَ مِن قَبْلِكُمْ قَوْمِ نُوحٍۢ وَعَادٍۢ وَثَمُودَ ۛ وَٱلَّذِينَ مِنۢ بَعْدِهِمْ ۛ لَا يَعْلَمُهُمْ إِلَّا ٱللَّهُ ۚ جَآءَتْهُمْ رُسُلُهُم بِٱلْبَيِّنَتِ فَرَدُّوٓا۟ أَيْدِيَهُمْ فِىٓ أَفْوَهِهِمْ وَقَالُوٓا۟ إِنَّا كَفَرْنَا بِمَآ أُرْسِلْتُم بِهِۦ وَإِنَّا لَفِى شَكٍّۢ مِّمَّا تَدْعُونَنَآ إِلَيْهِ مُرِيبٍۢ ‎</p>
 </td>
 <td width="50%">'ʔalam ya'ʔtikum nabawʔu' 'l:aðiyna min qablikum qawmi nuwḥinm waʕa'dinm waθamuwda  wa'l:aðiyna minm baʕdihim  la' yaʕlamuhum 'ʔil:a' 'll:ahu j ja'ʔʔathum rusuluhum bi'lbay:inati farad:uwʔ' 'ʔaydiyahum fiyʔ 'ʔafwahihim waqa'luwʔ' 'ʔin:a' kafarna' bima'ʔ 'ʔursiltum bihiy wa'ʔin:a' lafiy ʃak:inm m:im:a' tadʕuwnana'ʔ 'ʔilayhi muriybinm </td>
 </tr>

 </td>
 <td width="50%"> qa'lat rusuluhum 'ʔafiy 'll:ahi ʃak:unm fa'ṭiri 'ls:amawati wa'l'ʔarḍi sl' yadʕuwkum liyaɣfira lakum m:in ðunuwbikum wayuwʔax:irakum 'ʔilay'ʔ 'ʔajalinm m:usam:anmy j qa'luwʔ' 'ʔin 'ʔantum 'ʔil:a' baʃarunm m:iθluna' turiyduwna 'ʔan taṣud:uwna' ʕam:a' ka'na yaʕbudu ʔa'ba'ʔwʔuna' fa'ʔtuwna' bisulṭaninm m:ubiyninm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_014_011">11</a> ‏قَالَتْ لَهُمْ رُسُلُهُمْ إِن نَّحْنُ إِلَّا بَشَرٌۭ مِّثْلُكُمْ وَلَكِنَّ ٱللَّهَ يَمُنُّ عَلَىٰ مَن يَشَآءُ مِنْ عِبَادِهِۦ ۖ وَمَا كَانَ لَنَآ أَن نَّأْتِيَكُم بِسُلْطَنٍ إِلَّا بِإِذْنِ ٱللَّهِ ۚ وَعَلَى ٱللَّهِ فَلْيَتَوَكَّلِ ٱلْمُؤْمِنُونَ ‎</p>
 </td>
 <td width="50%">qa'lat lahum rusuluhum 'ʔin n:aḥnu 'ʔil:a' baʃarunm m:iθlukum walakin:a 'll:aha yamun:u ʕalay' man yaʃa'ʔʔu min ʕiba'dihiy sl' wama' ka'na lana'ʔ 'ʔan n:a'ʔtiyakum bisulṭanin 'ʔil:a' bi'ʔiðni 'll:ahi j waʕalay 'll:ahi falyatawak:ali 'lmuwʔminuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_014_012">12</a> ‏وَمَا لَنَآ أَلَّا نَتَوَكَّلَ عَلَى ٱللَّهِ وَقَدْ هَدَىٰنَا سُبُلَنَا ۚ وَلَنَصْبِرَنَّ عَلَىٰ مَآ ءَاذَيْتُمُونَا ۚ وَعَلَى ٱللَّهِ فَلْيَتَوَكَّلِ ٱلْمُتَوَكِّلُونَ ‎</p>

 <td width="50%">wama' lana'ʔ 'ʔal:a' natawak:ala ʕalay 'll:ahi waqad haday'na' subulana' j walanaṣbiran:a ʕalay' ma'ʔ ʔa'ðaytumuwna' j waʕalay 'll:ahi falyatawak:ali 'lmutawak:iluwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_014_013">13</a> ‏وَقَالَ ٱلَّذِينَ كَفَرُوا۟ لِرُسُلِهِمْ لَنُخْرِجَنَّكُم مِّنْ أَرْضِنَآ أَوْ لَتَعُودُنَّ فِى مِلَّتِنَا ۖ فَأَوْحَىٰٓ إِلَيْهِمْ رَبُّهُمْ لَنُهْلِكَنَّ ٱلظَّلِمِينَ ‎</p>

 <td width="50%">waqa'la 'l:aðiyna kafaruw' lirusulihim lanuxrijan:akum m:in 'ʔarḍina'ʔ 'ʔaw lataʕuwdun:a fiy mil:atina' sl' fa'ʔawḥay'ʔ 'ʔilayhim rab:uhum lanuhlikan:a 'lẓ:alimiyna </td>

 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_014_019">19</a> ‏أَلَمْ تَرَ أَنَّ ٱللَّهَ خَلَقَ ٱلسَّمَوَتِ وَٱلْأَرْضَ بِٱلْحَقِّ ۚ إِن يَشَأْ يُذْهِبْكُمْ وَيَأْتِ بِخَلْقٍۢ جَدِيدٍۢ ‎</p>
 </td>
 <td width="50%">'ʔalam tara 'ʔan:a 'll:aha xalaqa 'ls:amawati wa'l'ʔarḍa bi'lḥaq:i j 'ʔin yaʃa'ʔ yuðhibkum waya'ʔti bixalqinm jadiydinm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_014_020">20</a> ‏وَمَا ذَلِكَ عَلَى ٱللَّهِ بِعَزِيزٍۢ ‎</p>
 </td>
 <td width="50%">wama' ðalika ʕalay 'll:ahi biʕaziyzinm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_014_021">21</a> ‏وَبَرَزُوا۟ لِلَّهِ جَمِيعًۭا فَقَالَ ٱلضُّعَفَٓؤُا۟ لِلَّذِينَ ٱسْتَكْبَرُوٓا۟ إِنَّا كُنَّا لَكُمْ تَبَعًۭا فَهَلْ أَنتُم مُّغْنُونَ عَنَّا مِنْ عَذَابِ ٱللَّهِ مِن شَىْءٍۢ ۚ قَالُوا۟ لَوْ هَدَىٰنَا ٱللَّهُ لَهَدَيْنَكُمْ ۖ سَوَآءٌ عَلَيْنَآ أَجَزِعْنَآ أَمْ صَبَرْنَا مَا لَنَا مِن مَّحِيصٍۢ ‎</p>
 </td>
 <td width="50%">wabarazuw' lil:ahi jamiyʕanm' faqa'la 'lḍ:uʕafaʔwʔu' lil:aðiyna 'stakbaruwʔ' 'ʔin:a' kun:a' lakum tabaʕanm' fahal 'ʔantum m:uɣnuwna ʕan:a' min ʕaða'bi 'll:ahi min ʃayʔinm j qa'luw' law haday'na' 'll:ahu lahadaynakum sl' sawa'ʔʔun ʕalayna'ʔ 'ʔajaziʕna'ʔ 'ʔam ṣabarna' ma' lana' min m:aḥiyṣinm </td>
