 <h1 align="CENTER">7. ‏الأعراف‎</h1>
 <h2 align="CENTER">'l'ʔʕr'f</h2>
 <table width="100%" border="1">
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_001">1</a> ‏الٓمٓصٓ ‎</p>
 </td>
 <td width="50%">'lʔmʔṣʔ </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_002">2</a> ‏كِتَبٌ أُنزِلَ إِلَيْكَ فَلَا يَكُن فِى صَدْرِكَ حَرَجٌۭ مِّنْهُ لِتُنذِرَ بِهِۦ وَذِكْرَىٰ لِلْمُؤْمِنِينَ ‎</p>
 </td>
 <td width="50%">kitabun 'ʔunzila 'ʔilayka fala' yakun fiy ṣadrika ḥarajunm m:inhu litunðira bihiy waðikray' lilmuwʔminiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_003">3</a> ‏ٱتَّبِعُوا۟ مَآ أُنزِلَ إِلَيْكُم مِّن رَّبِّكُمْ وَلَا تَتَّبِعُوا۟ مِن دُونِهِۦٓ أَوْلِيَآءَ ۗ قَلِيلًۭا مَّا تَذَكَّرُونَ ‎</p>
 </td>
 <td width="50%">'t:abiʕuw' ma'ʔ 'ʔunzila 'ʔilaykum m:in r:ab:ikum wala' tat:abiʕuw' min duwnihiyʔ 'ʔawliya'ʔʔa ql' qaliylanm' m:a' taðak:aruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_004">4</a> ‏وَكَم مِّن قَرْيَةٍ أَهْلَكْنَهَا فَجَآءَهَا بَأْسُنَا بَيَتًا أَوْ هُمْ قَآئِلُونَ ‎</p>
 </td>
 <td width="50%">wakam m:in qaryatin 'ʔahlaknaha' faja'ʔʔaha' ba'ʔsuna' bayatan' 'ʔaw hum qa'ʔyʔiluwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_005">5</a> ‏فَمَا كَانَ دَعْوَىٰهُمْ إِذْ جَآءَهُم بَأْسُنَآ إِلَّآ أَن قَالُوٓا۟ إِنَّا كُنَّا ظَلِمِينَ ‎</p>
 </td>
 <td width="50%">fama' ka'na daʕway'hum 'ʔið ja'ʔʔahum ba'ʔsuna'ʔ 'ʔil:a'ʔ 'ʔan qa'luwʔ' 'ʔin:a' kun:a' ẓalimiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_006">6</a> ‏فَلَنَسْـَٔلَنَّ ٱلَّذِينَ أُرْسِلَ إِلَيْهِمْ وَلَنَسْـَٔلَنَّ ٱلْمُرْسَلِينَ ‎</p>
 </td>
 <td width="50%">falanasʔalan:a 'l:aðiyna 'ʔursila 'ʔilayhim walanasʔalan:a 'lmursaliyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_007">7</a> ‏فَلَنَقُصَّنَّ عَلَيْهِم بِعِلْمٍۢ ۖ وَمَا كُنَّا غَآئِبِينَ ‎</p>
 </td>
 <td width="50%">falanaquṣ:an:a ʕalayhim biʕilminm sl' wama' kun:a' ɣa'ʔyʔibiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_008">8</a> ‏وَٱلْوَزْنُ يَوْمَئِذٍ ٱلْحَقُّ ۚ فَمَن ثَقُلَتْ مَوَزِينُهُۥ فَأُو۟لَٓئِكَ هُمُ ٱلْمُفْلِحُونَ ‎</p>
 </td>
 <td width="50%">wa'lwaznu yawmayʔiðin 'lḥaq:u j faman θaqulat mawaziynuhuw fa'ʔuwlaʔyʔika humu 'lmufliḥuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_009">9</a> ‏وَمَنْ خَفَّتْ مَوَزِينُهُۥ فَأُو۟لَٓئِكَ ٱلَّذِينَ خَسِرُوٓا۟ أَنفُسَهُم بِمَا كَانُوا۟ بِـَٔايَتِنَا يَظْلِمُونَ ‎</p>
 </td>
 <td width="50%">waman xaf:at mawaziynuhuw fa'ʔuwlaʔyʔika 'l:aðiyna xasiruwʔ' 'ʔanfusahum bima' ka'nuw' biʔa'yatina' yaẓlimuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_010">10</a> ‏وَلَقَدْ مَكَّنَّكُمْ فِى ٱلْأَرْضِ وَجَعَلْنَا لَكُمْ فِيهَا مَعَيِشَ ۗ قَلِيلًۭا مَّا تَشْكُرُونَ ‎</p>
 </td>
 <td width="50%">walaqad mak:an:akum fiy 'l'ʔarḍi wajaʕalna' lakum fiyha' maʕayiʃa ql' qaliylanm' m:a' taʃkuruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_011">11</a> ‏وَلَقَدْ خَلَقْنَكُمْ ثُمَّ صَوَّرْنَكُمْ ثُمَّ قُلْنَا لِلْمَلَٓئِكَةِ ٱسْجُدُوا۟ لِءَادَمَ فَسَجَدُوٓا۟ إِلَّآ إِبْلِيسَ لَمْ يَكُن مِّنَ ٱلسَّجِدِينَ ‎</p>
 </td>
 <td width="50%">walaqad xalaqnakum θum:a ṣaw:arnakum θum:a qulna' lilmalaʔyʔikati 'sjuduw' liʔa'dama fasajaduwʔ' 'ʔil:a'ʔ 'ʔibliysa lam yakun m:ina 'ls:ajidiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_012">12</a> ‏قَالَ مَا مَنَعَكَ أَلَّا تَسْجُدَ إِذْ أَمَرْتُكَ ۖ قَالَ أَنَا۠ خَيْرٌۭ مِّنْهُ خَلَقْتَنِى مِن نَّارٍۢ وَخَلَقْتَهُۥ مِن طِينٍۢ ‎</p>
 </td>
 <td width="50%">qa'la ma' manaʕaka 'ʔal:a' tasjuda 'ʔið 'ʔamartuka sl' qa'la 'ʔana' xayrunm m:inhu xalaqtaniy min n:a'rinm waxalaqtahuw min ṭiyninm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_013">13</a> ‏قَالَ فَٱهْبِطْ مِنْهَا فَمَا يَكُونُ لَكَ أَن تَتَكَبَّرَ فِيهَا فَٱخْرُجْ إِنَّكَ مِنَ ٱلصَّغِرِينَ ‎</p>
 </td>
 <td width="50%">qa'la fa'hbiṭ minha' fama' yakuwnu laka 'ʔan tatakab:ara fiyha' fa'xruj 'ʔin:aka mina 'lṣ:aɣiriyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_014">14</a> ‏قَالَ أَنظِرْنِىٓ إِلَىٰ يَوْمِ يُبْعَثُونَ ‎</p>
 </td>
 <td width="50%">qa'la 'ʔanẓirniyʔ 'ʔilay' yawmi yubʕaθuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_015">15</a> ‏قَالَ إِنَّكَ مِنَ ٱلْمُنظَرِينَ ‎</p>
 </td>
 <td width="50%">qa'la 'ʔin:aka mina 'lmunẓariyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_016">16</a> ‏قَالَ فَبِمَآ أَغْوَيْتَنِى لَأَقْعُدَنَّ لَهُمْ صِرَطَكَ ٱلْمُسْتَقِيمَ ‎</p>
 </td>
 <td width="50%">qa'la fabima'ʔ 'ʔaɣwaytaniy la'ʔaqʕudan:a lahum ṣiraṭaka 'lmustaqiyma </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_017">17</a> ‏ثُمَّ لَءَاتِيَنَّهُم مِّنۢ بَيْنِ أَيْدِيهِمْ وَمِنْ خَلْفِهِمْ وَعَنْ أَيْمَنِهِمْ وَعَن شَمَآئِلِهِمْ ۖ وَلَا تَجِدُ أَكْثَرَهُمْ شَكِرِينَ ‎</p>
 </td>
 <td width="50%">θum:a laʔa'tiyan:ahum m:inm bayni 'ʔaydiyhim wamin xalfihim waʕan 'ʔaymanihim waʕan ʃama'ʔyʔilihim sl' wala' tajidu 'ʔakθarahum ʃakiriyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_018">18</a> ‏قَالَ ٱخْرُجْ مِنْهَا مَذْءُومًۭا مَّدْحُورًۭا ۖ لَّمَن تَبِعَكَ مِنْهُمْ لَأَمْلَأَنَّ جَهَنَّمَ مِنكُمْ أَجْمَعِينَ ‎</p>
 </td>
 <td width="50%">qa'la 'xruj minha' maðʔuwmanm' m:adḥuwranm' sl' l:aman tabiʕaka minhum la'ʔamla'ʔan:a jahan:ama minkum 'ʔajmaʕiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_019">19</a> ‏وَيَٓـَٔادَمُ ٱسْكُنْ أَنتَ وَزَوْجُكَ ٱلْجَنَّةَ فَكُلَا مِنْ حَيْثُ شِئْتُمَا وَلَا تَقْرَبَا هَذِهِ ٱلشَّجَرَةَ فَتَكُونَا مِنَ ٱلظَّلِمِينَ ‎</p>
 </td>
 <td width="50%">wayaʔʔa'damu 'skun 'ʔanta wazawjuka 'ljan:ata fakula' min ḥayθu ʃiyʔtuma' wala' taqraba' haðihi 'lʃ:ajarata fatakuwna' mina 'lẓ:alimiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_020">20</a> ‏فَوَسْوَسَ لَهُمَا ٱلشَّيْطَنُ لِيُبْدِىَ لَهُمَا مَا وُۥرِىَ عَنْهُمَا مِن سَوْءَتِهِمَا وَقَالَ مَا نَهَىٰكُمَا رَبُّكُمَا عَنْ هَذِهِ ٱلشَّجَرَةِ إِلَّآ أَن تَكُونَا مَلَكَيْنِ أَوْ تَكُونَا مِنَ ٱلْخَلِدِينَ ‎</p>
 </td>
 <td width="50%">fawaswasa lahuma' 'lʃ:ayṭanu liyubdiya lahuma' ma' wuwriya ʕanhuma' min sawʔatihima' waqa'la ma' nahay'kuma' rab:ukuma' ʕan haðihi 'lʃ:ajarati 'ʔil:a'ʔ 'ʔan takuwna' malakayni 'ʔaw takuwna' mina 'lxalidiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_021">21</a> ‏وَقَاسَمَهُمَآ إِنِّى لَكُمَا لَمِنَ ٱلنَّصِحِينَ ‎</p>
 </td>
 <td width="50%">waqa'samahuma'ʔ 'ʔin:iy lakuma' lamina 'ln:aṣiḥiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_022">22</a> ‏فَدَلَّىٰهُمَا بِغُرُورٍۢ ۚ فَلَمَّا ذَاقَا ٱلشَّجَرَةَ بَدَتْ لَهُمَا سَوْءَتُهُمَا وَطَفِقَا يَخْصِفَانِ عَلَيْهِمَا مِن وَرَقِ ٱلْجَنَّةِ ۖ وَنَادَىٰهُمَا رَبُّهُمَآ أَلَمْ أَنْهَكُمَا عَن تِلْكُمَا ٱلشَّجَرَةِ وَأَقُل لَّكُمَآ إِنَّ ٱلشَّيْطَنَ لَكُمَا عَدُوٌّۭ مُّبِينٌۭ ‎</p>
 </td>
 <td width="50%">fadal:ay'huma' biɣuruwrinm j falam:a' ða'qa' 'lʃ:ajarata badat lahuma' sawʔatuhuma' waṭafiqa' yaxṣifa'ni ʕalayhima' min waraqi 'ljan:ati sl' wana'day'huma' rab:uhuma'ʔ 'ʔalam 'ʔanhakuma' ʕan tilkuma' 'lʃ:ajarati wa'ʔaqul l:akuma'ʔ 'ʔin:a 'lʃ:ayṭana lakuma' ʕaduw:unm m:ubiynunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_023">23</a> ‏قَالَا رَبَّنَا ظَلَمْنَآ أَنفُسَنَا وَإِن لَّمْ تَغْفِرْ لَنَا وَتَرْحَمْنَا لَنَكُونَنَّ مِنَ ٱلْخَسِرِينَ ‎</p>
 </td>
 <td width="50%">qa'la' rab:ana' ẓalamna'ʔ 'ʔanfusana' wa'ʔin l:am taɣfir lana' watarḥamna' lanakuwnan:a mina 'lxasiriyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_024">24</a> ‏قَالَ ٱهْبِطُوا۟ بَعْضُكُمْ لِبَعْضٍ عَدُوٌّۭ ۖ وَلَكُمْ فِى ٱلْأَرْضِ مُسْتَقَرٌّۭ وَمَتَعٌ إِلَىٰ حِينٍۢ ‎</p>
 </td>
 <td width="50%">qa'la 'hbiṭuw' baʕḍukum libaʕḍin ʕaduw:unm sl' walakum fiy 'l'ʔarḍi mustaqar:unm wamataʕun 'ʔilay' ḥiyninm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_025">25</a> ‏قَالَ فِيهَا تَحْيَوْنَ وَفِيهَا تَمُوتُونَ وَمِنْهَا تُخْرَجُونَ ‎</p>
 </td>
 <td width="50%">qa'la fiyha' taḥyawna wafiyha' tamuwtuwna waminha' tuxrajuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_026">26</a> ‏يَبَنِىٓ ءَادَمَ قَدْ أَنزَلْنَا عَلَيْكُمْ لِبَاسًۭا يُوَرِى سَوْءَتِكُمْ وَرِيشًۭا ۖ وَلِبَاسُ ٱلتَّقْوَىٰ ذَلِكَ خَيْرٌۭ ۚ ذَلِكَ مِنْ ءَايَتِ ٱللَّهِ لَعَلَّهُمْ يَذَّكَّرُونَ ‎</p>
 </td>
 <td width="50%">yabaniyʔ ʔa'dama qad 'ʔanzalna' ʕalaykum liba'sanm' yuwariy sawʔatikum wariyʃanm' sl' waliba'su 'lt:aqway' ðalika xayrunm j ðalika min ʔa'yati 'll:ahi laʕal:ahum yað:ak:aruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_027">27</a> ‏يَبَنِىٓ ءَادَمَ لَا يَفْتِنَنَّكُمُ ٱلشَّيْطَنُ كَمَآ أَخْرَجَ أَبَوَيْكُم مِّنَ ٱلْجَنَّةِ يَنزِعُ عَنْهُمَا لِبَاسَهُمَا لِيُرِيَهُمَا سَوْءَتِهِمَآ ۗ إِنَّهُۥ يَرَىٰكُمْ هُوَ وَقَبِيلُهُۥ مِنْ حَيْثُ لَا تَرَوْنَهُمْ ۗ إِنَّا جَعَلْنَا ٱلشَّيَطِينَ أَوْلِيَآءَ لِلَّذِينَ لَا يُؤْمِنُونَ ‎</p>
 </td>
 <td width="50%">yabaniyʔ ʔa'dama la' yaftinan:akumu 'lʃ:ayṭanu kama'ʔ 'ʔaxraja 'ʔabawaykum m:ina 'ljan:ati yanziʕu ʕanhuma' liba'sahuma' liyuriyahuma' sawʔatihima'ʔ ql' 'ʔin:ahuw yaray'kum huwa waqabiyluhuw min ḥayθu la' tarawnahum ql' 'ʔin:a' jaʕalna' 'lʃ:ayaṭiyna 'ʔawliya'ʔʔa lil:aðiyna la' yuwʔminuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_028">28</a> ‏وَإِذَا فَعَلُوا۟ فَحِشَةًۭ قَالُوا۟ وَجَدْنَا عَلَيْهَآ ءَابَآءَنَا وَٱللَّهُ أَمَرَنَا بِهَا ۗ قُلْ إِنَّ ٱللَّهَ لَا يَأْمُرُ بِٱلْفَحْشَآءِ ۖ أَتَقُولُونَ عَلَى ٱللَّهِ مَا لَا تَعْلَمُونَ ‎</p>
 </td>
 <td width="50%">wa'ʔiða' faʕaluw' faḥiʃatanm qa'luw' wajadna' ʕalayha'ʔ ʔa'ba'ʔʔana' wa'll:ahu 'ʔamarana' biha' ql' qul 'ʔin:a 'll:aha la' ya'ʔmuru bi'lfaḥʃa'ʔʔi sl' 'ʔataquwluwna ʕalay 'll:ahi ma' la' taʕlamuwna </td>
 </tr>
 <tr valign="TOP">

 <td width="50%"> huwa 'l:aðiy xalaqakum m:in n:afsinm waḥidatinm wajaʕala minha' zawjaha' liyaskuna 'ʔilayha' sl' falam:a' taɣaʃ:ay'ha' ḥamalat ḥamlan' xafiyfanm' famar:at bihiy sl' falam:a'ʔ 'ʔaθqalat d:aʕawa' 'll:aha rab:ahuma' layʔin ʔa'taytana' ṣaliḥanm' l:anakuwnan:a mina 'lʃ:akiriyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_190">190</a> ‏فَلَمَّآ ءَاتَىٰهُمَا صَلِحًۭا جَعَلَا لَهُۥ شُرَكَآءَ فِيمَآ ءَاتَىٰهُمَا ۚ فَتَعَلَى ٱللَّهُ عَمَّا يُشْرِكُونَ ‎</p>
 </td>
 <td width="50%">falam:a'ʔ ʔa'tay'huma' ṣaliḥanm' jaʕala' lahuw ʃuraka'ʔʔa fiyma'ʔ ʔa'tay'huma' j fataʕalay 'll:ahu ʕam:a' yuʃrikuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_191">191</a> ‏أَيُشْرِكُونَ مَا لَا يَخْلُقُ شَيْـًۭٔا وَهُمْ يُخْلَقُونَ ‎</p>

 </tr>

 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_193">193</a> ‏وَإِن تَدْعُوهُمْ إِلَى ٱلْهُدَىٰ لَا يَتَّبِعُوكُمْ ۚ سَوَآءٌ عَلَيْكُمْ أَدَعَوْتُمُوهُمْ أَمْ أَنتُمْ صَمِتُونَ ‎</p>
 </td>
 <td width="50%">wa'ʔin tadʕuwhum 'ʔilay 'lhuday' la' yat:abiʕuwkum j sawa'ʔʔun ʕalaykum 'ʔadaʕawtumuwhum 'ʔam 'ʔantum ṣamituwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_194">194</a> ‏إِنَّ ٱلَّذِينَ تَدْعُونَ مِن دُونِ ٱللَّهِ عِبَادٌ أَمْثَالُكُمْ ۖ فَٱدْعُوهُمْ فَلْيَسْتَجِيبُوا۟ لَكُمْ إِن كُنتُمْ صَدِقِينَ ‎</p>
 </td>
 <td width="50%">'ʔin:a 'l:aðiyna tadʕuwna min duwni 'll:ahi ʕiba'dun 'ʔamθa'lukum sl' fa'dʕuwhum falyastajiybuw' lakum 'ʔin kuntum ṣadiqiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_195">195</a> ‏أَلَهُمْ أَرْجُلٌۭ يَمْشُونَ بِهَآ ۖ أَمْ لَهُمْ أَيْدٍۢ يَبْطِشُونَ بِهَآ ۖ أَمْ لَهُمْ أَعْيُنٌۭ يُبْصِرُونَ بِهَآ ۖ أَمْ لَهُمْ ءَاذَانٌۭ يَسْمَعُونَ بِهَا ۗ قُلِ ٱدْعُوا۟ شُرَكَآءَكُمْ ثُمَّ كِيدُونِ فَلَا تُنظِرُونِ ‎</p>
 </td>
 <td width="50%">'ʔalahum 'ʔarjulunm yamʃuwna biha'ʔ sl' 'ʔam lahum 'ʔaydinm yabṭiʃuwna biha'ʔ sl' 'ʔam lahum 'ʔaʕyununm yubṣiruwna biha'ʔ sl' 'ʔam lahum ʔa'ða'nunm yasmaʕuwna biha' ql' quli 'dʕuw' ʃuraka'ʔʔakum θum:a kiyduwni fala' tunẓiruwni </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_196">196</a> ‏إِنَّ وَلِۦِّىَ ٱللَّهُ ٱلَّذِى نَزَّلَ ٱلْكِتَبَ ۖ وَهُوَ يَتَوَلَّى ٱلصَّلِحِينَ ‎</p>
 </td>
 <td width="50%">'ʔin:a waliy:iya 'll:ahu 'l:aðiy naz:ala 'lkitaba sl' wahuwa yatawal:ay 'lṣ:aliḥiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_197">197</a> ‏وَٱلَّذِينَ تَدْعُونَ مِن دُونِهِۦ لَا يَسْتَطِيعُونَ نَصْرَكُمْ وَلَآ أَنفُسَهُمْ يَنصُرُونَ ‎</p>
 </td>
 <td width="50%">wa'l:aðiyna tadʕuwna min duwnihiy la' yastaṭiyʕuwna naṣrakum wala'ʔ 'ʔanfusahum yanṣuruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_198">198</a> ‏وَإِن تَدْعُوهُمْ إِلَى ٱلْهُدَىٰ لَا يَسْمَعُوا۟ ۖ وَتَرَىٰهُمْ يَنظُرُونَ إِلَيْكَ وَهُمْ لَا يُبْصِرُونَ ‎</p>
 </td>
 <td width="50%">wa'ʔin tadʕuwhum 'ʔilay 'lhuday' la' yasmaʕuw' sl' wataray'hum yanẓuruwna 'ʔilayka wahum la' yubṣiruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_199">199</a> ‏خُذِ ٱلْعَفْوَ وَأْمُرْ بِٱلْعُرْفِ وَأَعْرِضْ عَنِ ٱلْجَهِلِينَ ‎</p>
 </td>
 <td width="50%">xuði 'lʕafwa wa'ʔmur bi'lʕurfi wa'ʔaʕriḍ ʕani 'ljahiliyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_200">200</a> ‏وَإِمَّا يَنزَغَنَّكَ مِنَ ٱلشَّيْطَنِ نَزْغٌۭ فَٱسْتَعِذْ بِٱللَّهِ ۚ إِنَّهُۥ سَمِيعٌ عَلِيمٌ ‎</p>
 </td>
 <td width="50%">wa'ʔim:a' yanzaɣan:aka mina 'lʃ:ayṭani nazɣunm fa'staʕið bi'll:ahi j 'ʔin:ahuw samiyʕun ʕaliymun </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_201">201</a> ‏إِنَّ ٱلَّذِينَ ٱتَّقَوْا۟ إِذَا مَسَّهُمْ طَٓئِفٌۭ مِّنَ ٱلشَّيْطَنِ تَذَكَّرُوا۟ فَإِذَا هُم مُّبْصِرُونَ ‎</p>
 </td>
 <td width="50%">'ʔin:a 'l:aðiyna 't:aqaw' 'ʔiða' mas:ahum ṭaʔyʔifunm m:ina 'lʃ:ayṭani taðak:aruw' fa'ʔiða' hum m:ubṣiruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_202">202</a> ‏وَإِخْوَنُهُمْ يَمُدُّونَهُمْ فِى ٱلْغَىِّ ثُمَّ لَا يُقْصِرُونَ ‎</p>
 </td>
 <td width="50%">wa'ʔixwanuhum yamud:uwnahum fiy 'lɣay:i θum:a la' yuqṣiruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_007_203">203</a> ‏وَإِذَا لَمْ تَأْتِهِم بِـَٔايَةٍۢ قَالُوا۟ لَوْلَا ٱجْتَبَيْتَهَا ۚ قُلْ إِنَّمَآ أَتَّبِعُ مَا يُوحَىٰٓ إِلَىَّ مِن رَّبِّى ۚ هَذَا بَصَآئِرُ مِن رَّبِّكُمْ وَهُدًۭى وَرَحْمَةٌۭ لِّقَوْمٍۢ يُؤْمِنُونَ ‎</p>
 </td>

 </td>
 <td width="50%">wa'ʔiða' quriyʔa 'lqurʔa'nu fa'stamiʕuw' lahuw wa'ʔanṣituw' laʕal:akum turḥamuwna </td>
