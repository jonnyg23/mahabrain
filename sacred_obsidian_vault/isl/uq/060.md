 <h1 align="CENTER">60. ‏الممتحنة‎</h1>
 <h2 align="CENTER">'lmmtḥnt</h2>
 <table width="100%" border="1">
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_060_001">1</a> ‏يَٓأَيُّهَاء ٱلَّذِينَ ءَامَنُوا۟ لَا تَتَّخِذُوا۟ عَدُوِّى وَعَدُوَّكُمْ أَوْلِيَآءَ تُلْقُونَ إِلَيْهِم بِٱلْمَوَدَّةِ وَقَدْ كَفَرُوا۟ بِمَا جَآءَكُم مِّنَ ٱلْحَقِّ يُخْرِجُونَ ٱلرَّسُولَ وَإِيَّاكُمْ ۙ أَن تُؤْمِنُوا۟ بِٱللَّهِ رَبِّكُمْ إِن كُنتُمْ خَرَجْتُمْ جِهَدًۭا فِى سَبِيلِى وَٱبْتِغَآءَ مَرْضَاتِى ۚ تُسِرُّونَ إِلَيْهِم بِٱلْمَوَدَّةِ وَأَنَا۠ أَعْلَمُ بِمَآ أَخْفَيْتُمْ وَمَآ أَعْلَنتُمْ ۚ وَمَن يَفْعَلْهُ مِنكُمْ فَقَدْ ضَلَّ سَوَآءَ ٱلسَّبِيلِ ‎</p>
 </td>
 <td width="50%">yaʔ'ʔay:uha'ʔ 'l:aðiyna ʔa'manuw' la' tat:axiðuw' ʕaduw:iy waʕaduw:akum 'ʔawliya'ʔʔa tulquwna 'ʔilayhim bi'lmawad:ati waqad kafaruw' bima' ja'ʔʔakum m:ina 'lḥaq:i yuxrijuwna 'lr:asuwla wa'ʔiy:a'kum l' 'ʔan tuwʔminuw' bi'll:ahi rab:ikum 'ʔin kuntum xarajtum jihadanm' fiy sabiyliy wa'btiɣa'ʔʔa marḍa'tiy j tusir:uwna 'ʔilayhim bi'lmawad:ati wa'ʔana' 'ʔaʕlamu bima'ʔ 'ʔaxfaytum wama'ʔ 'ʔaʕlantum j waman yafʕalhu minkum faqad ḍal:a sawa'ʔʔa 'ls:abiyli </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_060_002">2</a> ‏إِن يَثْقَفُوكُمْ يَكُونُوا۟ لَكُمْ أَعْدَآءًۭ وَيَبْسُطُوٓا۟ إِلَيْكُمْ أَيْدِيَهُمْ وَأَلْسِنَتَهُم بِٱلسُّوٓءِ وَوَدُّوا۟ لَوْ تَكْفُرُونَ ‎</p>
 </td>
 <td width="50%">'ʔin yaθqafuwkum yakuwnuw' lakum 'ʔaʕda'ʔʔanm wayabsuṭuwʔ' 'ʔilaykum 'ʔaydiyahum wa'ʔalsinatahum bi'ls:uwʔʔi wawad:uw' law takfuruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_060_003">3</a> ‏لَن تَنفَعَكُمْ أَرْحَامُكُمْ وَلَآ أَوْلَدُكُمْ ۚ يَوْمَ ٱلْقِيَمَةِ يَفْصِلُ بَيْنَكُمْ ۚ وَٱللَّهُ بِمَا تَعْمَلُونَ بَصِيرٌۭ ‎</p>
 </td>
 <td width="50%">lan tanfaʕakum 'ʔarḥa'mukum wala'ʔ 'ʔawladukum j yawma 'lqiyamati yafṣilu baynakum j wa'll:ahu bima' taʕmaluwna baṣiyrunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_060_004">4</a> ‏قَدْ كَانَتْ لَكُمْ أُسْوَةٌ حَسَنَةٌۭ فِىٓ إِبْرَهِيمَ وَٱلَّذِينَ مَعَهُۥٓ إِذْ قَالُوا۟ لِقَوْمِهِمْ إِنَّا بُرَءَٓؤُا۟ مِنكُمْ وَمِمَّا تَعْبُدُونَ مِن دُونِ ٱللَّهِ كَفَرْنَا بِكُمْ وَبَدَا بَيْنَنَا وَبَيْنَكُمُ ٱلْعَدَوَةُ وَٱلْبَغْضَآءُ أَبَدًا حَتَّىٰ تُؤْمِنُوا۟ بِٱللَّهِ وَحْدَهُۥٓ إِلَّا قَوْلَ إِبْرَهِيمَ لِأَبِيهِ لَأَسْتَغْفِرَنَّ لَكَ وَمَآ أَمْلِكُ لَكَ مِنَ ٱللَّهِ مِن شَىْءٍۢ ۖ رَّبَّنَا عَلَيْكَ تَوَكَّلْنَا وَإِلَيْكَ أَنَبْنَا وَإِلَيْكَ ٱلْمَصِيرُ ‎</p>
 </td>
 <td width="50%">qad ka'nat lakum 'ʔuswatun ḥasanatunm fiyʔ 'ʔibrahiyma wa'l:aðiyna maʕahuwʔ 'ʔið qa'luw' liqawmihim 'ʔin:a' buraʔaʔwʔu' minkum wamim:a' taʕbuduwna min duwni 'll:ahi kafarna' bikum wabada' baynana' wabaynakumu 'lʕadawatu wa'lbaɣḍa'ʔʔu 'ʔabadan' ḥat:ay' tuwʔminuw' bi'll:ahi waḥdahuwʔ 'ʔil:a' qawla 'ʔibrahiyma li'ʔabiyhi la'ʔastaɣfiran:a laka wama'ʔ 'ʔamliku laka mina 'll:ahi min ʃayʔinm sl' r:ab:ana' ʕalayka tawak:alna' wa'ʔilayka 'ʔanabna' wa'ʔilayka 'lmaṣiyru </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_060_005">5</a> ‏رَبَّنَا لَا تَجْعَلْنَا فِتْنَةًۭ لِّلَّذِينَ كَفَرُوا۟ وَٱغْفِرْ لَنَا رَبَّنَآ ۖ إِنَّكَ أَنتَ ٱلْعَزِيزُ ٱلْحَكِيمُ ‎</p>
 </td>
 <td width="50%">rab:ana' la' tajʕalna' fitnatanm l:il:aðiyna kafaruw' wa'ɣfir lana' rab:ana'ʔ sl' 'ʔin:aka 'ʔanta 'lʕaziyzu 'lḥakiymu </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_060_006">6</a> ‏لَقَدْ كَانَ لَكُمْ فِيهِمْ أُسْوَةٌ حَسَنَةٌۭ لِّمَن كَانَ يَرْجُوا۟ ٱللَّهَ وَٱلْيَوْمَ ٱلْءَاخِرَ ۚ وَمَن يَتَوَلَّ فَإِنَّ ٱللَّهَ هُوَ ٱلْغَنِىُّ ٱلْحَمِيدُ ‎</p>
 </td>
 <td width="50%">laqad ka'na lakum fiyhim 'ʔuswatun ḥasanatunm l:iman ka'na yarjuw' 'll:aha wa'lyawma 'lʔa'xira j waman yatawal:a fa'ʔin:a 'll:aha huwa 'lɣaniy:u 'lḥamiydu </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_060_007">7</a> ‏ عَسَى ٱللَّهُ أَن يَجْعَلَ بَيْنَكُمْ وَبَيْنَ ٱلَّذِينَ عَادَيْتُم مِّنْهُم مَّوَدَّةًۭ ۚ وَٱللَّهُ قَدِيرٌۭ ۚ وَٱللَّهُ غَفُورٌۭ رَّحِيمٌۭ ‎</p>
 </td>
 <td width="50%"> ʕasay 'll:ahu 'ʔan yajʕala baynakum wabayna 'l:aðiyna ʕa'daytum m:inhum m:awad:atanm j wa'll:ahu qadiyrunm j wa'll:ahu ɣafuwrunm r:aḥiymunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_060_008">8</a> ‏لَّا يَنْهَىٰكُمُ ٱللَّهُ عَنِ ٱلَّذِينَ لَمْ يُقَتِلُوكُمْ فِى ٱلدِّينِ وَلَمْ يُخْرِجُوكُم مِّن دِيَرِكُمْ أَن تَبَرُّوهُمْ وَتُقْسِطُوٓا۟ إِلَيْهِمْ ۚ إِنَّ ٱللَّهَ يُحِبُّ ٱلْمُقْسِطِينَ ‎</p>
 </td>
 <td width="50%">l:a' yanhay'kumu 'll:ahu ʕani 'l:aðiyna lam yuqatiluwkum fiy 'ld:iyni walam yuxrijuwkum m:in diyarikum 'ʔan tabar:uwhum watuqsiṭuwʔ' 'ʔilayhim j 'ʔin:a 'll:aha yuḥib:u 'lmuqsiṭiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_060_009">9</a> ‏إِنَّمَا يَنْهَىٰكُمُ ٱللَّهُ عَنِ ٱلَّذِينَ قَتَلُوكُمْ فِى ٱلدِّينِ وَأَخْرَجُوكُم مِّن دِيَرِكُمْ وَظَهَرُوا۟ عَلَىٰٓ إِخْرَاجِكُمْ أَن تَوَلَّوْهُمْ ۚ وَمَن يَتَوَلَّهُمْ فَأُو۟لَٓئِكَ هُمُ ٱلظَّلِمُونَ ‎</p>
 </td>
 <td width="50%">'ʔin:ama' yanhay'kumu 'll:ahu ʕani 'l:aðiyna qataluwkum fiy 'ld:iyni wa'ʔaxrajuwkum m:in diyarikum waẓaharuw' ʕalay'ʔ 'ʔixra'jikum 'ʔan tawal:awhum j waman yatawal:ahum fa'ʔuwlaʔyʔika humu 'lẓ:alimuwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_060_010">10</a> ‏يَٓأَيُّهَا ٱلَّذِينَ ءَامَنُوٓا۟ إِذَا جَآءَكُمُ ٱلْمُؤْمِنَتُ مُهَجِرَتٍۢ فَٱمْتَحِنُوهُنَّ ۖ ٱللَّهُ أَعْلَمُ بِإِيمَنِهِنَّ ۖ فَإِنْ عَلِمْتُمُوهُنَّ مُؤْمِنَتٍۢ فَلَا تَرْجِعُوهُنَّ إِلَى ٱلْكُفَّارِ ۖ لَا هُنَّ حِلٌّۭ لَّهُمْ وَلَا هُمْ يَحِلُّونَ لَهُنَّ ۖ وَءَاتُوهُم مَّآ أَنفَقُوا۟ ۚ وَلَا جُنَاحَ عَلَيْكُمْ أَن تَنكِحُوهُنَّ إِذَآ ءَاتَيْتُمُوهُنَّ أُجُورَهُنَّ ۚ وَلَا تُمْسِكُوا۟ بِعِصَمِ ٱلْكَوَافِرِ وَسْـَٔلُوا۟ مَآ أَنفَقْتُمْ وَلْيَسْـَٔلُوا۟ مَآ أَنفَقُوا۟ ۚ ذَلِكُمْ حُكْمُ ٱللَّهِ ۖ يَحْكُمُ بَيْنَكُمْ ۚ وَٱللَّهُ عَلِيمٌ حَكِيمٌۭ ‎</p>
 </td>
 <td width="50%">yaʔ'ʔay:uha' 'l:aðiyna ʔa'manuwʔ' 'ʔiða' ja'ʔʔakumu 'lmuwʔminatu muhajiratinm fa'mtaḥinuwhun:a sl' 'll:ahu 'ʔaʕlamu bi'ʔiymanihin:a sl' fa'ʔin ʕalimtumuwhun:a muwʔminatinm fala' tarjiʕuwhun:a 'ʔilay 'lkuf:a'ri sl' la' hun:a ḥil:unm l:ahum wala' hum yaḥil:uwna lahun:a sl' waʔa'tuwhum m:a'ʔ 'ʔanfaquw' j wala' juna'ḥa ʕalaykum 'ʔan tankiḥuwhun:a 'ʔiða'ʔ ʔa'taytumuwhun:a 'ʔujuwrahun:a j wala' tumsikuw' biʕiṣami 'lkawa'firi wasʔaluw' ma'ʔ 'ʔanfaqtum walyasʔaluw' ma'ʔ 'ʔanfaquw' j ðalikum ḥukmu 'll:ahi sl' yaḥkumu baynakum j wa'll:ahu ʕaliymun ḥakiymunm </td>
