 <h1 align="CENTER">66. ‏التحريم‎</h1>
 <h2 align="CENTER">'ltḥrym</h2>
 <table width="100%" border="1">
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_066_001">1</a> ‏يَٓأَيُّهَا ٱلنَّبِىُّ لِمَ تُحَرِّمُ مَآ أَحَلَّ ٱللَّهُ لَكَ ۖ تَبْتَغِى مَرْضَاتَ أَزْوَجِكَ ۚ وَٱللَّهُ غَفُورٌۭ رَّحِيمٌۭ ‎</p>
 </td>
 <td width="50%">yaʔ'ʔay:uha' 'ln:abiy:u lima tuḥar:imu ma'ʔ 'ʔaḥal:a 'll:ahu laka sl' tabtaɣiy marḍa'ta 'ʔazwajika j wa'll:ahu ɣafuwrunm r:aḥiymunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_066_002">2</a> ‏قَدْ فَرَضَ ٱللَّهُ لَكُمْ تَحِلَّةَ أَيْمَنِكُمْ ۚ وَٱللَّهُ مَوْلَىٰكُمْ ۖ وَهُوَ ٱلْعَلِيمُ ٱلْحَكِيمُ ‎</p>
 </td>
 <td width="50%">qad faraḍa 'll:ahu lakum taḥil:ata 'ʔaymanikum j wa'll:ahu mawlay'kum sl' wahuwa 'lʕaliymu 'lḥakiymu </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_066_003">3</a> ‏وَإِذْ أَسَرَّ ٱلنَّبِىُّ إِلَىٰ بَعْضِ أَزْوَجِهِۦ حَدِيثًۭا فَلَمَّا نَبَّأَتْ بِهِۦ وَأَظْهَرَهُ ٱللَّهُ عَلَيْهِ عَرَّفَ بَعْضَهُۥ وَأَعْرَضَ عَنۢ بَعْضٍۢ ۖ فَلَمَّا نَبَّأَهَا بِهِۦ قَالَتْ مَنْ أَنۢبَأَكَ هَذَا ۖ قَالَ نَبَّأَنِىَ ٱلْعَلِيمُ ٱلْخَبِيرُ ‎</p>
 </td>
 <td width="50%">wa'ʔið 'ʔasar:a 'ln:abiy:u 'ʔilay' baʕḍi 'ʔazwajihiy ḥadiyθanm' falam:a' nab:a'ʔat bihiy wa'ʔaẓharahu 'll:ahu ʕalayhi ʕar:afa baʕḍahuw wa'ʔaʕraḍa ʕanm baʕḍinm sl' falam:a' nab:a'ʔaha' bihiy qa'lat man 'ʔanmba'ʔaka haða' sl' qa'la nab:a'ʔaniya 'lʕaliymu 'lxabiyru </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_066_004">4</a> ‏إِن تَتُوبَآ إِلَى ٱللَّهِ فَقَدْ صَغَتْ قُلُوبُكُمَا ۖ وَإِن تَظَهَرَا عَلَيْهِ فَإِنَّ ٱللَّهَ هُوَ مَوْلَىٰهُ وَجِبْرِيلُ وَصَلِحُ ٱلْمُؤْمِنِينَ ۖ وَٱلْمَلَٓئِكَةُ بَعْدَ ذَلِكَ ظَهِيرٌ ‎</p>
 </td>
 <td width="50%">'ʔin tatuwba'ʔ 'ʔilay 'll:ahi faqad ṣaɣat quluwbukuma' sl' wa'ʔin taẓahara' ʕalayhi fa'ʔin:a 'll:aha huwa mawlay'hu wajibriylu waṣaliḥu 'lmuwʔminiyna sl' wa'lmalaʔyʔikatu baʕda ðalika ẓahiyrun </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_066_005">5</a> ‏عَسَىٰ رَبُّهُۥٓ إِن طَلَّقَكُنَّ أَن يُبْدِلَهُۥٓ أَزْوَجًا خَيْرًۭا مِّنكُنَّ مُسْلِمَتٍۢ مُّؤْمِنَتٍۢ قَنِتَتٍۢ تَٓئِبَتٍ عَبِدَتٍۢ سَٓئِحَتٍۢ ثَيِّبَتٍۢ وَأَبْكَارًۭا ‎</p>
 </td>
 <td width="50%">ʕasay' rab:uhuwʔ 'ʔin ṭal:aqakun:a 'ʔan yubdilahuwʔ 'ʔazwajan' xayranm' m:inkun:a muslimatinm m:uwʔminatinm qanitatinm taʔyʔibatin ʕabidatinm saʔyʔiḥatinm θay:ibatinm wa'ʔabka'ranm' </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_066_006">6</a> ‏يَٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ قُوٓا۟ أَنفُسَكُمْ وَأَهْلِيكُمْ نَارًۭا وَقُودُهَا ٱلنَّاسُ وَٱلْحِجَارَةُ عَلَيْهَا مَلَٓئِكَةٌ غِلَاظٌۭ شِدَادٌۭ لَّا يَعْصُونَ ٱللَّهَ مَآ أَمَرَهُمْ وَيَفْعَلُونَ مَا يُؤْمَرُونَ ‎</p>
 </td>
 <td width="50%">yaʔ'ʔay:uha' 'l:aðiyna ʔa'manuw' quwʔ' 'ʔanfusakum wa'ʔahliykum na'ranm' waquwduha' 'ln:a'su wa'lḥija'ratu ʕalayha' malaʔyʔikatun ɣila'ẓunm ʃida'dunm l:a' yaʕṣuwna 'll:aha ma'ʔ 'ʔamarahum wayafʕaluwna ma' yuwʔmaruwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_066_007">7</a> ‏يَٓأَيُّهَا ٱلَّذِينَ كَفَرُوا۟ لَا تَعْتَذِرُوا۟ ٱلْيَوْمَ ۖ إِنَّمَا تُجْزَوْنَ مَا كُنتُمْ تَعْمَلُونَ ‎</p>
 </td>
 <td width="50%">yaʔ'ʔay:uha' 'l:aðiyna kafaruw' la' taʕtaðiruw' 'lyawma sl' 'ʔin:ama' tujzawna ma' kuntum taʕmaluwna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_066_008">8</a> ‏يَٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ تُوبُوٓا۟ إِلَى ٱللَّهِ تَوْبَةًۭ نَّصُوحًا عَسَىٰ رَبُّكُمْ أَن يُكَفِّرَ عَنكُمْ سَيِّـَٔاتِكُمْ وَيُدْخِلَكُمْ جَنَّتٍۢ تَجْرِى مِن تَحْتِهَا ٱلْأَنْهَرُ يَوْمَ لَا يُخْزِى ٱللَّهُ ٱلنَّبِىَّ وَٱلَّذِينَ ءَامَنُوا۟ مَعَهُۥ ۖ نُورُهُمْ يَسْعَىٰ بَيْنَ أَيْدِيهِمْ وَبِأَيْمَنِهِمْ يَقُولُونَ رَبَّنَآ أَتْمِمْ لَنَا نُورَنَا وَٱغْفِرْ لَنَآ ۖ إِنَّكَ عَلَىٰ كُلِّ شَىْءٍۢ قَدِيرٌۭ ‎</p>
 </td>
 <td width="50%">yaʔ'ʔay:uha' 'l:aðiyna ʔa'manuw' tuwbuwʔ' 'ʔilay 'll:ahi tawbatanm n:aṣuwḥan' ʕasay' rab:ukum 'ʔan yukaf:ira ʕankum say:iʔa'tikum wayudxilakum jan:atinm tajriy min taḥtiha' 'l'ʔanharu yawma la' yuxziy 'll:ahu 'ln:abiy:a wa'l:aðiyna ʔa'manuw' maʕahuw sl' nuwruhum yasʕay' bayna 'ʔaydiyhim wabi'ʔaymanihim yaquwluwna rab:ana'ʔ 'ʔatmim lana' nuwrana' wa'ɣfir lana'ʔ sl' 'ʔin:aka ʕalay' kul:i ʃayʔinm qadiyrunm </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_066_009">9</a> ‏يَٓأَيُّهَا ٱلنَّبِىُّ جَهِدِ ٱلْكُفَّارَ وَٱلْمُنَفِقِينَ وَٱغْلُظْ عَلَيْهِمْ ۚ وَمَأْوَىٰهُمْ جَهَنَّمُ ۖ وَبِئْسَ ٱلْمَصِيرُ ‎</p>
 </td>
 <td width="50%">yaʔ'ʔay:uha' 'ln:abiy:u jahidi 'lkuf:a'ra wa'lmunafiqiyna wa'ɣluẓ ʕalayhim j wama'ʔway'hum jahan:amu sl' wabiyʔsa 'lmaṣiyru </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_066_010">10</a> ‏ضَرَبَ ٱللَّهُ مَثَلًۭا لِّلَّذِينَ كَفَرُوا۟ ٱمْرَأَتَ نُوحٍۢ وَٱمْرَأَتَ لُوطٍۢ ۖ كَانَتَا تَحْتَ عَبْدَيْنِ مِنْ عِبَادِنَا صَلِحَيْنِ فَخَانَتَاهُمَا فَلَمْ يُغْنِيَا عَنْهُمَا مِنَ ٱللَّهِ شَيْـًۭٔا وَقِيلَ ٱدْخُلَا ٱلنَّارَ مَعَ ٱلدَّخِلِينَ ‎</p>
 </td>
 <td width="50%">ḍaraba 'll:ahu maθalanm' l:il:aðiyna kafaruw' 'mra'ʔata nuwḥinm wa'mra'ʔata luwṭinm sl' ka'nata' taḥta ʕabdayni min ʕiba'dina' ṣaliḥayni faxa'nata'huma' falam yuɣniya' ʕanhuma' mina 'll:ahi ʃayʔanm' waqiyla 'dxula' 'ln:a'ra maʕa 'ld:axiliyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_066_011">11</a> ‏وَضَرَبَ ٱللَّهُ مَثَلًۭا لِّلَّذِينَ ءَامَنُوا۟ ٱمْرَأَتَ فِرْعَوْنَ إِذْ قَالَتْ رَبِّ ٱبْنِ لِى عِندَكَ بَيْتًۭا فِى ٱلْجَنَّةِ وَنَجِّنِى مِن فِرْعَوْنَ وَعَمَلِهِۦ وَنَجِّنِى مِنَ ٱلْقَوْمِ ٱلظَّلِمِينَ ‎</p>
 </td>
 <td width="50%">waḍaraba 'll:ahu maθalanm' l:il:aðiyna ʔa'manuw' 'mra'ʔata firʕawna 'ʔið qa'lat rab:i 'bni liy ʕindaka baytanm' fiy 'ljan:ati wanaj:iniy min firʕawna waʕamalihiy wanaj:iniy mina 'lqawmi 'lẓ:alimiyna </td>
 </tr>
 <tr valign="TOP">
 <td width="50%"><p align="RIGHT"><a name="an_066_012">12</a> ‏وَمَرْيَمَ ٱبْنَتَ عِمْرَنَ ٱلَّتِىٓ أَحْصَنَتْ فَرْجَهَا فَنَفَخْنَا فِيهِ مِن رُّوحِنَا وَصَدَّقَتْ بِكَلِمَتِ رَبِّهَا وَكُتُبِهِۦ وَكَانَتْ مِنَ ٱلْقَنِتِينَ ‎</p>
 </td>
 <td width="50%">wamaryama 'bnata ʕimrana 'l:atiyʔ 'ʔaḥṣanat farjaha' fanafaxna' fiyhi min r:uwḥina' waṣad:aqat bikalimati rab:iha' wakutubihiy waka'nat mina 'lqanitiyna </td>
 </tr>
